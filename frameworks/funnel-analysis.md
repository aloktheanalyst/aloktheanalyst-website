coaching:
  use_for: "e-commerce funnel, conversion optimization, drop-off analysis, user journey."
  steps:
    - title: DEFINE FUNNEL STAGES
      prompt: |
        Define the funnel stages clearly (e.g., Landing -> Browse -> Add to Cart -> Checkout -> Payment -> Confirmation)
    - title: CONVERSION RATES
      prompt: |
        Calculate conversion rate at each step
    - title: BIGGEST DROP-OFF
      prompt: |
        Identify the biggest drop-off point
    - title: SEGMENT
      prompt: |
        Segment by dimensions: device, channel, user type (new vs returning), geography
    - title: HYPOTHESIZE
      prompt: |
        Hypothesize why drop-off happens at that stage
    - title: PROPOSE SOLUTIONS
      prompt: |
        Propose solutions: UX fix, A/B test, personalization
    - title: SQL
      prompt: |
        Write queries to compute funnel metrics from event tables
